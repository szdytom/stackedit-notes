ARC121E Directed Trees题解

题目相当于要求 iii 不能在点 iii 的后代的节点上出现。

如果直接求满足所有条件的树的个数是不大好求的，我们可以先将它转化一下，求不满足条件的树的个数。记 fi,jf_{i,j}fi,j​ 表示在从点 iii 出发能到达的所有点构成的子图上选择 jjj 个点不满足条件（剩下的暂时不做决定）的方案数，显然有

fu,j=∑(u→v)∈E∑k=0jfv,kfu,j−kf_{u,j}=\sum_{(u\to v)\in \mathrm E}\sum_{k=0}^jf_{v,k}f_{u,j-k}fu,j​=(u→v)∈E∑​k=0∑j​fv,k​fu,j−k​

统计答案时则使用容斥原理，

∑i=0n(−1)if1,i(n−i)!\sum_{i=0}^n(-1)^if_{1,i}(n-i)!i=0∑n​(−1)if1,i​(n−i)!

乘上 (n−i)!(n-i)!(n−i)! 代表除了已经不合格的点以外其它点的选择方式，但由于其它点随便选也有可能导致产生更多的点不合格，这就是为什么我们要进行容斥。时间复杂度 O(n2)\mathcal O(n^2)O(n2)。

参考实现：

#include <cstdio>
#include <vector>
#include <functional>
using namespace std;

const int mod = 998244353;

void add(int& u, int v) {
	u += v;
	if (u >= mod) u -= mod;
}

int main() {
	int n;
	scanf("%d", &n);
	vector<vector<int>> g(n + 1);
	for (int i = 2; i <= n; i++) {
		int p;
		scanf("%d", &p);
		g[p].push_back(i);
	}
	vector<int> son(n + 1);
	vector<vector<int>> f(n + 1, vector<int>(n + 1, 0));
	
	function<void(int)> dfs = [&](int u) {
		son[u] = 1;
		for (int v: g[u]) {
			dfs(v);
			son[u] += son[v];
		}
		f[u][0] = 1;
		int upper = 0;
		for (int v: g[u]) {
			vector<int> t(upper + son[v] + 1, 0);
			for (int x = 0; x <= upper; x++) {
				if (f[u][x] != 0) {
					for (int y = 0; y <= son[v]; y++) {
						add(t[x + y], f[u][x] *1ll* f[v][y] % mod);
					}
				}
			}
			upper += son[v];
			for (int i = 0; i <= upper; i++) f[u][i] = t[i];
		}
		for (int i = son[u]; i >= 0; i--) {
			int left = son[u] - 1 - i;
			if (left >= 1) {
				add(f[u][i + 1], f[u][i] *1ll* left % mod);
			}
		}
	};
	
	dfs(1);
	vector<int> p(n + 1);
	p[0] = 1;
	for (int i = 1; i <= n; i++) p[i] = p[i - 1] * 1ll * i % mod;
	int ans = 0;
	for (int i = 0; i <= n; i++) {
		int way = f[1][i] * 1ll * p[n - i] % mod;
		if (i % 2) way = way * 1ll * (mod - 1) % mod;
		add(ans, way);
	}
	printf("%d\n", ans);
	return 0;	
}
