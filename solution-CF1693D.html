CF1693D Decinc Dividing题解

若存在一段合法区间 [l..r][l..r][l..r]，那么 ∀l⩽i⩽j⩽r\forall_{l\leqslant i \leqslant j \leqslant r}∀l⩽i⩽j⩽r​，有 [i..j][i..j][i..j] 是合法区间，这样就证明了答案有单调性，即对于某个位置 iii，定义 wiw_iwi​ 是最大的满足 [i..wi][i..w_i][i..wi​] 为合法区间的正整数，那么有 wi⩾wi−1w_i\geqslant w_{i-1}wi​⩾wi−1​。

考虑类似整体二分的做法，对于区间 [l..r][l..r][l..r]，我们先求出 wlw_lwl​ 和 wrw_rwr​，若 wl=wrw_l=w_rwl​=wr​，则将满足 l⩽i⩽rl\leqslant i\leqslant rl⩽i⩽r 的 wiw_iwi​ 都赋值为 wlw_lwl​，否则设 m=⌊l+r2⌋m=\lfloor \frac{l+r}{2} \rfloorm=⌊2l+r​⌋ 并递归处理 [l..m][l..m][l..m] 和 [m..r][m..r][m..r]。最后的答案即为 (∑i=1nwi)−n(n−1)2\left(\sum\limits_{i=1}^n w_i\right)-\frac{n(n-1)}{2}(i=1∑n​wi​)−2n(n−1)​。

参考实现：

#include <bits/stdc++.h>
using namespace std;

const int maxn = 2e5 + 5;

template <typename T> inline bool ckmin(T &x, const T y) { return x > y ? (x = y) || true : false; }
template <typename T> inline bool ckmax(T &x, const T y) { return x < y ? (x = y) || true : false; }

int a[maxn], ans[maxn], f[maxn][2], n;

inline
void dp(int k){
    f[k][0] = n + 1;
    f[k][1] = 0;
    for (int i = k + 1; i <= n; ++i) {
        f[i][0] = 0;
        f[i][1] = n + 1;

        if (a[i - 1] < a[i]) f[i][0] = f[i - 1][0];
        if (a[i - 1] > a[i]) f[i][1] = f[i - 1][1];
        if (f[i - 1][1] < a[i]) ckmax(f[i][0], a[i - 1]);
        if (f[i - 1][0] > a[i]) ckmin(f[i][1], a[i - 1]);
        if (f[i][0] == 0 && f[i][1] == n + 1) {
            ans[k] = i - 1;
            return;
        }
    }
    ans[k] = n;
}

void solve(int l,int r){
    if (l + 1 >= r) return;
    if (ans[l] == ans[r]){
	    for(int i = l; i <= r; ++i) ans[i] = ans[l];
	    return;
	}

    int mid = (l + r) / 2;
    dp(mid);
    solve(l,mid);
    solve(mid,r);
}

int main(){
    scanf("%d",&n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d",a+i);
    }

    dp(1);
    dp(n);
    solve(1,n);
    printf("%lld\n", accumulate(ans + 1, ans + n + 1, 0LL) - 1LL * n * (n - 1) / 2);
    return 0;
}
